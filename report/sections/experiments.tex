\chapter{Experiments}

\section{Quality Measures}

{Quality Measures}
\begin{itemize}
    \item Build time (minutes)
    \item Search latency (minutes)
    \item Index size (GB)
    \item Recall
\end{itemize}


\begin{definition}[Recall]
    Let \(\mathcal{K}\) be groundtruth, \(\mathcal{N}\) be ANNS output. Recall is
    \[
        \frac{|\mathcal{K} \cap \mathcal{N}|}{|\mathcal{K}|}
    \]
\end{definition}
